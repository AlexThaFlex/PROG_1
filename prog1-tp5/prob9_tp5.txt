int soma_binario(int x, int y){
  int sum[20], i=0, remainder = 0, num;
  while (x != 0 || y != 0)
    {
        sum =(x % 10 ^ y % 10 ^ remainder);
        remainder = ((x % 10)*(y % 10) + remainder * (x % 10 ^ y % 10));
        x = x / 10;
        y = y / 10;
        
        num = num * 10 + sum;

        return num;
    }
}
